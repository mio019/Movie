# 项目目录结构

movie_recommend/
├── app.py                  # Flask 主入口
├── config.py               # 配置文件
├── requirements.txt        # 依赖列表
├── init_db.py              # 数据库初始化脚本
├── import_data.py          # MovieLens 数据导入脚本
│
├── database/
│   └── movie_recommend.db  # SQLite 数据库文件（运行后生成）
│
├── data/                   # MovieLens 100K 原始数据
│   ├── u.data              # 评分数据
│   ├── u.item              # 电影数据
│   ├── u.user              # 用户数据
│   └── u.genre             # 类型数据
│
├── algorithms/             # 推荐算法模块
│   ├── __init__.py
│   ├── user_cf.py          # User-CF
│   ├── item_cf.py          # Item-CF
│   ├── svd_cf.py           # SVD矩阵分解
│   ├── content_based.py    # 基于内容过滤
│   ├── hybrid.py           # 混合推荐
│   └── evaluator.py        # 算法评估（RMSE/精确率/召回率）
│
├── routes/                 # 路由/接口模块
│   ├── __init__.py
│   ├── auth.py             # 注册/登录接口
│   ├── movies.py           # 电影相关接口
│   ├── user.py             # 用户相关接口
│   ├── recommend.py        # 推荐接口
│   └── admin.py            # 管理员接口
│
├── models/                 # 数据模型
│   ├── __init__.py
│   └── db.py               # 数据库连接与操作
│
├── static/                 # 静态资源
│   ├── css/
│   ├── js/
│   └── images/
│
└── templates/              # HTML 模板
    ├── user/               # 用户端页面
    │   ├── index.html
    │   ├── login.html
    │   ├── register.html
    │   ├── movies.html
    │   ├── movie_detail.html
    │   ├── profile.html
    │   └── recommend.html
    └── admin/              # 管理员端页面
        ├── dashboard.html
        ├── users.html
        ├── movies.html
        └── ratings.html
